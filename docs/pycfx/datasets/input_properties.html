 18    def __init__(self, 
 19                 feature_names: List[str], 
 20                 feature_classes: List[Literal['categorical', 'ordinal', 'numeric', 'ordinal_normalised']], 
 21                 bounds: List[Union[Optional[Tuple[float, float]], List[float]]],
 22                 n_targets: int,
 23                 categorical_groups: Union[List[List[int]], Literal['auto']] = 'auto',
 24                 y_onehot: bool=True
 25                ):
 26        
 27        """
 28        Initialise with a list of feature names, each of which has a corresponding feature class (categorical, ordinal, numeric, ordinal_normalised).
 29        Specify bounds for each: None for categorical, (min, max) for numeric, or a set of valid values for ordinal features.
 30        Specify the number of labels in `n_targets`, and whether these are onehot in `y_onehot`. 
 31        Specify indices of categorical features in categorical_groups. If not specified, these will be inferred from the feature names. 
 32        In this case, eature names with a common suffix followed by a _ will be treated as part of the same categorical group
 33        """
 34        self.n_targets = n_targets
 35
 36        assert len(feature_names) == len(feature_classes), "Elements in feature_names must correspond to elements in feature_classes"
 37        self.feature_names = feature_names
 38        self.feature_classes = feature_classes
 39        self.n_features = len(feature_names)
 40
 41        assert len(feature_names) == len(bounds), "Elements in feature_names must correspond to elements in bounds. Set elements to None for no bound or one-hot categorical"
 42        for i, bound in enumerate(bounds):
 43            if feature_classes[i] == 'ordinal':
 44                assert bound != None, "Ordinal features must have a lower and upper bound"
 45                assert (isinstance(bound[0], int) and isinstance(bound[1], int)) or (bound[0].is_integer() and bound[1].is_integer()), "Ordinal features must have an integer LB and UB"
 46            elif feature_classes[i] == 'ordinal_normalised':
 47                assert isinstance(bound, list) and all(0 <= b <= 1 for b in bound), "Ordinal normalised features must have bounds as a list of numbers between 0 and 1"
 48            if bound != None:
 49                assert bound[0] <= bound[1], "LB must be less than UB for each element"
 50
 51        self.bound = bounds
 52
 53        if categorical_groups == "auto":
 54            categorical_indices = [i for i in range(len(feature_classes)) if feature_classes[i] == 'categorical']
 55            categorical_sets = defaultdict(list)
 56            for index in categorical_indices:
 57                category_name = feature_names[index].split('_')[0]
 58                categorical_sets[category_name].append(index)
 59            self.categorical_groups = list(categorical_sets.values())
 60        else:
 61            for group in categorical_groups:
 62                for index in group:
 63                    assert index > 0 and index < len(feature_names), f"Index {index} of group {group} is out of bounds"
 64                    assert feature_classes[index] == 'categorical', f"Index {index} of group {group} is not a categorical feature"
 65        
 66            self.categorical_groups = categorical_groups  
 67
 68        self.y_onehot = y_onehot
 69        self.n_distinct_features = self.n_features - sum([len(g) for g in self.categorical_groups]) + len(self.categorical_groups)
 70        self.all_idx = np.arange(self.n_features)
 71
 72        if len(self.categorical_groups) == 0:
 73            self.all_cat_idx = np.array([])
 74            self.non_cat_idx = self.all_idx
 75        else:
 76            self.all_cat_idx = np.concatenate(self.categorical_groups)
 77            self.non_cat_idx = self.all_idx[~np.isin(self.all_idx, self.all_cat_idx)]

 84        for i in range(self.n_features):
 85            feature_details.append([self.feature_names[i], self.feature_classes[i], self.bound[i]])
 86        return feature_details

 90        Get a list of dataset labels

 92        return list(range(self.n_targets))

 94    def check_valid_instance(self, x: np.ndarray) -> None:
 95        """
 96        Check if `x` is a valid (properly encoded) instance.
 97        """
 98
 99        for i in range(self.n_features):
100            feature_class = self.feature_classes[i]
101            bound = self.bound[i]
102            feature_val = x[i]
103
104            if feature_class == 'numeric' and bound is not None:
105                assert feature_val >= bound[0] and feature_val <= bound[1], "Numeric feature {i} out of bounds"
106            elif feature_class == 'ordinal' or feature_class == 'ordinal_normalised':
107                assert np.any(np.isclose(feature_val, bound)), "Ordinal feature not of allowed values"
108        
109        for group in self.categorical_groups:
110            group_vals = x[group]
111            assert np.sum(group_vals == 0) == len(group) - 1, "Incorrectly encoded categorical variable"
112            assert np.sum(group_vals == 1) == 1, "Incorrectly encoded categorical variable"

114    def fix_encoding(self, x: np.ndarray) -> np.array:
115        """
116        Fix the encoding of `x` to the closest valid instance.
117        """
118
119        for i in range(self.n_features):
120            feature_class = self.feature_classes[i]
121            bound = self.bound[i]
122
123            if feature_class == 'numeric' and bound is not None:
124                if isinstance(x, np.ndarray):
125                    if x[i] <= bound[0]:
126                        x[i] = bound[0]
127                    elif x[i] >= bound[1]:
128                        x[i] = bound[1]
129
130            if feature_class == 'ordinal' or feature_class == 'ordinal_normalised':
131                if not np.any(np.isclose(x[i], bound)):
132                    val = bound[np.argmin(np.abs(np.array(bound) - x[i]))]
133                    x[i] = val
134
135        for group in self.categorical_groups:
136            group_vals = x[group]
137            closest_to_one = np.argmin(np.abs(group_vals - 1))
138            group_vals.fill(0)
139            group_vals[closest_to_one] = 1
140
141            x[group] = group_vals 
142        
143        return x

146        """
147        Print a human-readable description of a datapoint x, using feature names etc.. TODO: undo any preprocessing done by the dataset .describe_dataset() method.
148        """

152        for i in range(self.n_features):
153            if self.feature_classes[i] != 'categorical':

155

157            index_active = group[np.where(x[group] == 1)[0][0]]
158            print(f"categorical {self.feature_names[index_active]}")

186    def gp_set_input_var_constraints(self, grb_model: gp.Model) -> Tuple[list[gp.Var], gp.MVar]:
187        """
188        Given a `grb_model`, output (equivalent) Input Vars and an Input MVar, of length n_features, which are constrained to be correctly encoded.
189        """
190         
191        input_vars = []
192
193        input_mvar = grb_model.addMVar(
194            shape=(self.n_features,),
195            vtype=GRB.CONTINUOUS,
196            lb=-float('inf'),
197            ub=float('inf'),
198            name=f"inp_var"
199        )
200        
201        for i, feature in enumerate(self.feature_names):
202            feature_class = self.feature_classes[i]
203            bound = self.bound[i]
204            lb = -float('inf') 
205            ub = float('inf') 
206
207            if (feature_class == 'numeric' or feature_class == 'ordinal') and bound != None: 
208                lb = bound[0]
209                ub = bound[1]
210            elif feature_class == 'categorical' or feature_class == 'ordinal_normalised':
211                lb = 0
212                ub = 1
213
214            vtype = GRB.CONTINUOUS if feature_class == 'numeric' else GRB.INTEGER
215
216            if feature_class == 'ordinal_normalised':
217                chosen_value = grb_model.addMVar(shape=(len(bound),), vtype=GRB.BINARY, name=feature+"_picker")
218                grb_model.addConstr(gp.quicksum(chosen_value) == 1)
219                grb_model.addConstr(input_mvar[i] == gp.quicksum(bound[k] * chosen_value[k] for k in range(len(bound))))
220                input_vars.append(chosen_value)
221
222            else:
223                var = grb_model.addVar(lb=lb, ub=ub, vtype=vtype, name=feature)
224                grb_model.addConstr(input_mvar[i] == var)
225                input_vars.append(var)
226
227        for group in self.categorical_groups:
228            group_vars = [input_vars[i] for i in group]
229            grb_model.addConstr(gp.quicksum(group_vars) == 1)
230
231        return input_vars, input_mvar
