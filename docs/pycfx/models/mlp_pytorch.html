 22class PyTorchModel(DifferentiableModel):
 23    """
 24    Abstract wrapper for building, training, saving, loading and evaluating any PyTorch model
 25    """
 26
 27    def __init__(self, config: dict, input_properties: InputProperties):
 28        """
 29        Initialise with dataset `input_properties` and `config`
 30        The config key `random_state` is set to a fixed 0, hidden_dims set to [50], batch size to 64, epochs to 50, and learning rate to 0.01
 31        """
 32         
 33        super().__init__(config, input_properties)
 34        self.batch_size = self.config.get('batch_size', 64)
 35        self.epochs = self.config.get('epochs', 50)
 36        self.lr = self.config.get("lr", 0.01)
 37        self.device = torch.device("cuda" if torch.cuda.is_available() else "mps" if torch.mps.is_available() else "cpu")
 38
 39        if input_properties is not None:
 40            self.pytorch_model, self.loss_fn, self.optimiser = self._build_model()
 41            self.pytorch_model.to(self.device)
 42
 43    @abstractmethod
 44    def _build_model(self):
 45        """
 46        Use self.config and self.device, and return a tuple of (PyTorch model, PyTorch loss function and PyTorch optimiser)
 47        """
 48        pass
 49
 50    def train(self, X_train: np.ndarray, y_train: np.ndarray):
 51        torch.manual_seed(self.random_state)
 52        cfg = self.config
 53
 54        X_train_tensor = torch.tensor(X_train, dtype=torch.float32).to(self.device)
 55        y_train_tensor = torch.tensor(y_train, dtype=torch.long).to(self.device)
 56
 57        if self.input_properties.y_onehot:
 58            n_classes = self.input_properties.n_targets
 59            y_train_onehot = nn.functional.one_hot(y_train_tensor, num_classes=n_classes).float()
 60            train_dataset = TensorDataset(X_train_tensor, y_train_onehot)
 61        else:
 62            train_dataset = TensorDataset(X_train_tensor, y_train_tensor)
 63        
 64        train_loader = DataLoader(train_dataset, batch_size=self.batch_size, shuffle=True)
 65
 66        self.pytorch_model.train()
 67
 68        for epoch in range(self.epochs):
 69            for X_batch, y_batch in train_loader:
 70                X_batch = X_batch.to(self.device)
 71                y_batch = y_batch.to(self.device)
 72                self.optimiser.zero_grad()
 73                outputs = self.pytorch_model(X_batch)
 74                loss = self.loss_fn(outputs, y_batch)
 75                loss.backward()
 76                self.optimiser.step()
 77
 78        return self.pytorch_model
 79
 80    def predict(self, x: np.ndarray) -> np.array:
 81        return self.pytorch_model(torch.tensor(x, dtype=torch.float).to(self.device)).detach().cpu().numpy()
 82
 83    def load(self, save_path: Path) -> None:
 84        self.pytorch_model = torch.load(save_path, weights_only=False)
 85        self.pytorch_model.to(self.device)
 86        self.save_path = save_path
 87
 88    def save(self, save_path: Path):
 89        torch.save(self.pytorch_model, save_path)
 90        self.save_path = save_path
 91    
 92    def evaluate(self, X_test: np.ndarray, y_test: np.ndarray) -> dict:
 93        self.pytorch_model.eval()
 94
 95        X_test_tensor = torch.tensor(X_test, dtype=torch.float32).to(self.device)
 96        y_test_tensor = torch.tensor(y_test, dtype=torch.long).to(self.device)
 97
 98        with torch.no_grad():
 99            test_outputs = self.pytorch_model(X_test_tensor)
100            test_loss = self.loss_fn(test_outputs, y_test_tensor).item()
101            _, predicted = torch.max(test_outputs, 1)
102            accuracy = (predicted == y_test_tensor).sum().item() / y_test_tensor.size(0)
103            precision = precision_score(y_test_tensor.cpu().numpy(), predicted.cpu().numpy(), average='weighted')
104            f1 = f1_score(y_test_tensor.cpu().numpy(), predicted.cpu().numpy(), average='weighted')
105
106        model_performance = {
107            'loss': test_loss,
108            'accuracy': accuracy * 100,
109            'precision': precision * 100,
110            'f1_score': f1 * 100
111        }
112
113        return model_performance

116        return self.loss_fn(y1, y2)

124class PyTorchMLP(PyTorchModel, MILPEncodableModel):
125    """
126    Multi-layer perceptron using PyTorchModel

130    def _build_model(self) -> tuple[nn.Sequential, nn.CrossEntropyLoss, optim.Optimizer]:
131        if self.config.get("dims"):
132            self.dims = self.config["dims"]
133        else:
134            self.dims = self.config.get('hidden_dims', [50])
135            self.dims = [self.input_properties.n_features] + self.dims + [self.input_properties.n_targets]
136
137        layers = []
138
139        for i in range(len(self.dims) - 1):
140            layers.append(nn.Linear(self.dims[i], self.dims[i + 1]))
141            layers.append(nn.ReLU())
142
143        layers = layers[:-1]
144        
145        model = nn.Sequential(*layers)
146        model.to(self.device)
147
148        loss_fn = nn.CrossEntropyLoss()
149        optimiser = optim.Adam(model.parameters(), lr=self.lr)
150
151        return model, loss_fn, optimiser
152    
153    def gp_set_model_constraints(self, grb_model: gp.Model, input_mvar: gp.MVar) -> gp.MVar:
154        output_mvar = grb_model.addMVar(shape=(self.pytorch_model[-1].out_features,), lb=-GRB.INFINITY, ub=GRB.INFINITY, name="output")
155        add_sequential_constr(grb_model, self.pytorch_model.cpu(), input_mvar, output_mvar)
156        self.pytorch_model.to(self.device)
157        return output_mvar
158    
159    def gp_set_classification_constraint(self, grb_model: gp.Model, output_vars: gp.MVar, target_class: int, db_distance=1e-6) -> None:
160        classification_constrs = []
161
162        if output_vars.shape[0] == 1:
163            #Single output
164            assert target_class in [0, 1], "Target class must be 0 or 1 for a single logit output"
165
166            if target_class == 0:
167                c1 = grb_model.addConstr(output_vars[0] >= 0.5, name="Output class 0 constraint")
168                classification_constrs.append(c1)
169            else:
170                c2 = grb_model.addConstr(output_vars[0] <= 0.5, name="Output class 1 constraint")
171                classification_constrs.append(c2)
172            
173        else:
174            #One-hot output
175            for i in range(output_vars.shape[0]):
176                if i != target_class:
177                    c = grb_model.addConstr(output_vars[target_class] >= output_vars[i] + db_distance)
178                    classification_constrs.append(c)
179
180        
181        return classification_constrs
