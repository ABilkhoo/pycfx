208    def evaluate_counterfactuals(self, aggregate_means: bool=False) -> List[pd.DataFrame]:
209        """
210        6. Evaluate generated counterfactual explanations using the specified metrics. 
211        Saves JSON results to save_dir / evaluation_raw.json, and a LaTeX table to save_dir / evaluation_table.txt
212        """
213
214        factuals_output = self.set_factuals()
215        counterfactuals_output = self.get_counterfactuals()
216
217        model_generator_metrics = {}
218
219        for model_factory in self.model_factories:
220            for model_name, model_set in model_factory.get_models_over_repeats().items():
221                generator_metrics = defaultdict(list)
222                
223                for model in model_set:
224                    counterfactuals = counterfactuals_output[str(model.save_dir)]
225                    factuals_bank = factuals_output[str(model.save_dir)]
226                    factuals_bank = {k: (np.array(v[0]), np.array(v[1])) for k, v in factuals_bank.items()}
227
228
229                    for generator_name, counterfactuals_bank in counterfactuals.items():
230                        if generator_metrics.get(generator_name) is None:
231                            generator_metrics[generator_name] = defaultdict(list)
232
233                        for metric in self.metrics:
234                            metric_results = metric.compute_metric(model, self.dataset.input_properties, self.dataset, factuals_bank, counterfactuals_bank)
235                            if isinstance(metric_results, np.ndarray) and not aggregate_means:
236                                mean_result = np.nanmean(metric_results)    
237                                generator_metrics[generator_name][metric.name()].append(mean_result)
238                            else:
239                                generator_metrics[generator_name][metric.name()].append(metric_results.tolist())
240
241                combined_metrics = defaultdict(dict)
242                for generator, generator_results in generator_metrics.items():
243                    for metric, metric_results in generator_results.items():
244                        combined_metrics[generator][metric] = {
245                            "mean": np.mean(metric_results),
246                            "sd": np.std(metric_results)
247                        }
248
249                model_generator_metrics[str(model_name)] = {"raw": generator_metrics, "aggregated": combined_metrics}

274    def generate_table(self, 
275                       formatted_col_names=None, structure: list[str]=[], indent_map: dict[str, list[str]]={}, format_map: dict[str, str]={}, generators_predicate=lambda gen: True,
276                       distance_metric=None, plausibility_metric=None, implausibility_metric=None, sensitivity_metric=None, stability_metric=None,
277                       validity_metric=None, failures_metric=None, report_failures=True, report_invalidity=True, dp2: bool=False, scaling: list[tuple[int, float]]=[],
278                       include_extra=[]) -> List[pd.DataFrame]:
279        
280        """
281        Generate a formatted DataFrame and LaTeX table with this benchmarker's results. Saves to evaluation_table_2.txt.
282
283        Parameters:
284        - formatted_col_names: Column names of the table (i.e. metric names)
285        - structure: The order in which generators should be presented. 
286        - indent_map: config properties that should be displayed indented on the table. Supports up to 2 layers of identation
287        - format_map: used to identify raw generator names that map to display names used in structure and indent_map
288        - *_metric: specify the metric name for the particular metric. Otherwise will pick the first matching metric
289        - report_failures, report_invalidity: if True, will add lines to the text file detailing failures and invalidity
290        - dp2: round to 2dp.
291        - scaling: specify list of column indices and corresponding scaling factor. E.g. [(3, 0.1)] to scale 3rd column by 10%
292
293        For example usage, see results/results_notebook.ipynb
294        """
295        
296        if not self.eval_path_raw.is_file():
297            print("Run evaluate_counterfactuals first!")
298            return
299        
300        with open(self.eval_path_raw) as f:
301            model_generator_metrics = json.load(f)
302
303
304        metrics_objs = {
305            "distance_metric": distance_metric or next((m.name() for m in self.metrics if isinstance(m, DistanceMetric)), None),
306            "plausibility_metric": plausibility_metric or next((m.name() for m in self.metrics if isinstance(m, LOFMetric)), None),
307            "implausibility_metric": implausibility_metric or next((m.name() for m in self.metrics if isinstance(m, ImplausibilityMetric)), None),
308            "sensitivity_metric": sensitivity_metric or next((m.name() for m in self.metrics if isinstance(m, SensitivityMetric)), None),
309            "stability_metric": stability_metric or next((m.name() for m in self.metrics if isinstance(m, StabilityMetric)), None),
310            "failures_metric": failures_metric or next((m.name() for m in self.metrics if isinstance(m, FailuresMetric)), None),
311            "validity_metric": validity_metric or next((m.name() for m in self.metrics if isinstance(m, ValidityMetric)), None)
312        }
313
314        metrics_cols = [metrics_objs['distance_metric'], metrics_objs['plausibility_metric'], metrics_objs['implausibility_metric'], metrics_objs['sensitivity_metric'], metrics_objs['stability_metric']]
315        metrics_cols = [col for col in metrics_cols if col is not None]
316        metrics_cols += include_extra
317
318        if formatted_col_names is None or len(formatted_col_names) != len(metrics_cols):
319            formatted_col_names = metrics_cols
320
321        full_formatted_columns = ["__generator"] + formatted_col_names
322
323        rows_c = []
324        validity_strings = []
325
326        for model_type, metrics in model_generator_metrics.items():
327            df = pd.DataFrame(metrics['aggregated']).T
328            #Extract validity here
329            model_name = Path(model_type).name
330
331            if "Failures" in df and report_failures:
332                failures = df['Failures'].map(lambda x: x['mean'])
333                failures_str = json.dumps(dict(failures[failures >= 0.01]))
334                validity_strings.append(f"{model_name}, Failures " + failures_str)
335
336            if "Validity" in df and report_invalidity:
337                validity = df['Validity'].map(lambda x: x['mean'])
338                validity_str = json.dumps(dict(validity[validity <= 0.9]))
339                validity_strings.append(f"{model_name}, Validity " + validity_str)
340
341            df = df.loc[df.index[df.index.map(generators_predicate)], metrics_cols]
342            df.columns = formatted_col_names
343
344            for col_num, scaling_factor in scaling:
345                col_name = df.columns[col_num]
346                df[col_name] = df[col_name].apply(
347                    lambda x: {"mean": x["mean"] * scaling_factor, "sd": x["sd"] * scaling_factor} if isinstance(x, dict) else x
348                )
349            
350            rows_c.append(pd.DataFrame([{"__generator": f"\\textbf{{{model_name}}}"}], columns = full_formatted_columns))
351            rows_c.extend(self.format_table_with_structure(df, structure, indent_map, format_map, full_formatted_columns))
352                    
353
354        concat = pd.concat(rows_c)
355        for element in indent_map.values():
356            for v in element:
357                if v in concat.columns:
358                    del concat[v]
359
360        concat.index = concat["__generator"]
361        del concat["__generator"]
362
363        if dp2:
364            concat = concat.applymap(lambda x: f"{x['mean']:.2f} ± {x['sd']:.2f}" if isinstance(x, dict) else "")
365        else:
366            concat = concat.applymap(lambda x: f"{x['mean']:.3f} ± {x['sd']:.3f}" if isinstance(x, dict) else "")
367
368        concat = concat.reset_index()
369        concat = concat.rename(columns={"__generator":"Generator"})
370
371        with open(self.eval_path_table_2, "w") as f:
372            f.write(f"Date: {pd.Timestamp.now().strftime('%Y-%m-%d %H:%M:%S')}\n")
373            f.write(concat.to_latex(index=False))
374            f.write("\n")
375            f.write("\n".join(validity_strings))

688    def format_table_with_structure(self, df: pd.DataFrame, structure: list[str], indent_map: dict[str, list[str]], format_map: dict[str, str], columns: list[str]):
689        """
690        Helper to generate DataFrame rows to be concatenated into a formatted table following structure, indent_map and format_map
691
692        Parameters:
693        - structure: The order in which SplitConformalPrediction (+subclasses) instances should be presented. 
694        - indent_map: config properties that should be displayed indented on the table. Supports up to 2 layers of identation
695        - format_map: used to identify raw generator names that map to display names used in structure and indent_map
696        - columns: names of columns in df
697
698        """
699        rows_c = []
700
701        for label in structure:
702            df_label = format_map.get(label, label)
703            if df_label is None:
704                continue
705
706            rows = df[df.index.str.contains(df_label)]
707
708            if len(rows) == 0:
709                continue
710
711            indent_config = indent_map.get(label)
712
713            if indent_config is None:
714                rows.loc[:, "__generator"] = label
715                rows_c.append(rows)
716            elif indent_config == "config":
717                rows_c.append(rows)
718
719            elif len(indent_config) == 1:
720                indent_config = indent_config[0]
721                sep_row = pd.DataFrame([{"__generator": label}], columns = columns)
722
723                formatted_index_config = format_map.get(indent_config, indent_config)
724                
725                regex = f'"{formatted_index_config}"' + r':(\d+\.?\d*)'
726                rows.loc[:, indent_config] = rows.index.to_series().apply(lambda x: re.search(regex, x).group(1) if re.search(regex, x) else None)
727
728                rows = rows.sort_values(by=indent_config)
729
730                rows.loc[:, "__generator"] = f'$\\ \\ {indent_config} = ' + rows[indent_config].astype(str) + "$"
731                
732                rows_c.append(sep_row)
733                rows_c.append(rows)
734            elif len(indent_config) == 2:
735                indent_config_formatted = list(map(lambda x: format_map.get(x, x), indent_config))
736
737                for i, element in enumerate(indent_config):
738                    element_formatted = indent_config_formatted[i]
739                    regex = f'"{element_formatted}"' + r':(\d+\.?\d*)'
740                    rows.loc[:, element] = rows.index.to_series().apply(lambda x: re.search(regex, x).group(1) if re.search(regex, x) else None)
741
742                for indent_val, indent_rows in rows.groupby(indent_config[0], sort=True):
743                    sep_row = pd.DataFrame([{"__generator": f"{label}, ${indent_config[0]} = {indent_val}$"}], columns = columns)
744
745                    indent_rows = indent_rows.sort_values(by=indent_config[1])
746                        
747                    indent_rows.loc[:, "__generator"] = f'$\\ \\ {indent_config[1]} = ' + indent_rows[indent_config[1]].astype(str) + "$"
748
749                    rows_c.append(sep_row)
750                    rows_c.append(indent_rows)
751
752            else:
753                raise NotImplementedError("Max supported index depth is 2")
754            
755        return rows_c

689    def format_table_with_structure(self, df: pd.DataFrame, structure: list[str], indent_map: dict[str, list[str]], format_map: dict[str, str], columns: list[str]):
690        """
691        Helper to generate DataFrame rows to be concatenated into a formatted table following structure, indent_map and format_map
692
693        Parameters:
694        - structure: The order in which SplitConformalPrediction (+subclasses) instances should be presented. 
695        - indent_map: config properties that should be displayed indented on the table. Supports up to 2 layers of identation
696        - format_map: used to identify raw generator names that map to display names used in structure and indent_map
697        - columns: names of columns in df
698
699        """
700        rows_c = []
701
702        for label in structure:
703            df_label = format_map.get(label, label)
704            if df_label is None:
705                continue
706
707            rows = df[df.index.str.contains(df_label)]
708
709            if len(rows) == 0:
710                continue
711
712            indent_config = indent_map.get(label)
713
714            if indent_config is None:
715                rows.loc[:, "__generator"] = label
716                rows_c.append(rows)
717            elif indent_config == "config":
718                rows_c.append(rows)
719
720            elif len(indent_config) == 1:
721                indent_config = indent_config[0]
722                sep_row = pd.DataFrame([{"__generator": label}], columns = columns)
723
724                formatted_index_config = format_map.get(indent_config, indent_config)
725                
726                regex = f'"{formatted_index_config}"' + r':(\d+\.?\d*)'
727                rows.loc[:, indent_config] = rows.index.to_series().apply(lambda x: re.search(regex, x).group(1) if re.search(regex, x) else None)
728
729                rows = rows.sort_values(by=indent_config)
730
731                rows.loc[:, "__generator"] = f'$\\ \\ {indent_config} = ' + rows[indent_config].astype(str) + "$"
732                
733                rows_c.append(sep_row)
734                rows_c.append(rows)
735            elif len(indent_config) == 2:
736                indent_config_formatted = list(map(lambda x: format_map.get(x, x), indent_config))
737
738                for i, element in enumerate(indent_config):
739                    element_formatted = indent_config_formatted[i]
740                    regex = f'"{element_formatted}"' + r':(\d+\.?\d*)'
741                    rows.loc[:, element] = rows.index.to_series().apply(lambda x: re.search(regex, x).group(1) if re.search(regex, x) else None)
742
743                for indent_val, indent_rows in rows.groupby(indent_config[0], sort=True):
744                    sep_row = pd.DataFrame([{"__generator": f"{label}, ${indent_config[0]} = {indent_val}$"}], columns = columns)
745
746                    indent_rows = indent_rows.sort_values(by=indent_config[1])
747                        
748                    indent_rows.loc[:, "__generator"] = f'$\\ \\ {indent_config[1]} = ' + indent_rows[indent_config[1]].astype(str) + "$"
749
750                    rows_c.append(sep_row)
751                    rows_c.append(indent_rows)
752
753            else:
754                raise NotImplementedError("Max supported index depth is 2")
755            
756        return rows_c
